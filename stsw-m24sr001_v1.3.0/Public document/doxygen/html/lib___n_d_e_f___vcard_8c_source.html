61 {
62  u8* pLastByteAdd, *pLook4Word, *pEndString ;
63 
64  /* First charactere force to NULL in case not matching found */
65  *pString = 0;
66

74 
75  pLook4Word = pPayload;

91  {
92  memcpy( pString, pLook4Word, pEndString-pLook4Word);

94  pString += pEndString-pLook4Word;
