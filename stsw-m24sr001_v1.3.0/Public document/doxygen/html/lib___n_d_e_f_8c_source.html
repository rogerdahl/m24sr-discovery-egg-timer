302  u32 SPPayloadSize = 0;
303  u32 OffsetInSPPayload =0;
304  u32 RecordPosition = 0;
305  sRecordInfo *pSPRecordStruct;
306 
307  /* initialize variable with size of the payload and poiter on data */
308  PayloadSize = ((u32)(pRecordStruct->PayloadLength3)<<24) | ((u32)(pRecordStruct->PayloadLength2)<<16) |
309  ((u32)(pRecordStruct->PayloadLength1)<<8) | pRecordStruct->PayloadLength0;
310 
311  pPayload = (u8*)(pRecordStruct->PayloadBufferAdd);
312 
313  pSPRecordStruct = (sRecordInfo *)SPRecordStructAdd[0];
