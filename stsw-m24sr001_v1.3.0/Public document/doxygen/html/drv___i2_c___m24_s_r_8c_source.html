53 static int8_t M24SR_PollI2C ( void )
54 {
55 uint16_t M24SR_Timeout = 0,
56  NbAtempt = M24SR_I2C_POLLING,
57  uSR1register= 0x0000;
58 
59  /*---------------------------- Transmission Phase ---------------------------*/
60  do{
61  M24SR_Timeout = 0x40;
63  I2C_ClearFlag(M24SR_I2C, I2C_FLAG_AF);
64 
66  I2C_AcknowledgeConfig(M24SR_I2C, DISABLE);
67 
69  I2C_GenerateSTART(M24SR_I2C, ENABLE);
70 
72  while ((!I2C_GetFlagStatus(M24SR_I2C,I2C_FLAG_SB)) && --M24SR_Timeout)
73  {
74  }
75 
77  I2C_Send7bitAddress(M24SR_I2C, M24SR_ADDR, I2C_Direction_Transmitter);
78  M24SR_Timeout = 0x40;
79 
80 
81  uSR1register = I2C_ReadRegister(M24SR_I2C, I2C_Register_SR1);
82  /* Note: ADDR is not set after a NACK reception */
83  while ( ((uSR1register != 0x0400) && ((uSR1register&0x0002) !=0x0002)))/*BUSY & MSL & ADDR */
84  {
85  /*reads the SR1 register */
86  uSR1register = I2C_ReadRegister(M24SR_I2C, I2C_Register_SR1);
87  uSR1register &= 0x0482;
88  }
89 
90  /* Send STOP Condition */
91  I2C_GenerateSTOP(M24SR_I2C, ENABLE);
92 
93  if (uSR1register == 0x0002)
94  {
95  /* Disable the START condition generation */
96  I2C_Cmd(M24SR_I2C, DISABLE);
97  do
98  {
99  uSR1register = I2C_ReadRegister(M24SR_I2C, I2C_Register_SR1);
100  }
101  while( (uSR1register&0x0002) ==0x0002);
102 
103  I2C_Cmd(M24SR_I2C, ENABLE);
104 
105  return M24SR_STATUS_SUCCESS;
106  }
107 
108  } while ( (uSR1register != 0x0000) && (--NbAtempt > 0) );
109 
110  if (NbAtempt == 0)
111  {
112  return M24SR_ERROR_NBATEMPT;
113  }
114  return M24SR_ERROR_NACK;
115 
116 }

467 int8_t M24SR_IsAnswerReady ( void )
468 {
469  int8_t status;
470  uint32_t retry = 0x3FFFF;
471  uint8_t stable = 0;
472 
473 
474  switch (uSynchroMode)
475  {
476  case M24SR_WAITINGTIME_POLLING :
477  errchk(M24SR_PollI2C ( ));
478  return M24SR_STATUS_SUCCESS;
479 
480  case M24SR_WAITINGTIME_TIMEOUT :
481  delay_ms (10);
482  return M24SR_STATUS_SUCCESS;
483 
484  case M24SR_WAITINGTIME_GPO :
485  /* Here we check if I2C answer is ready, so if GPO of M24SR is low */
486  do
487  {
488  if( GPIO_ReadInputDataBit(M24SR_GPO_PIN_PORT,M24SR_GPO_PIN) == Bit_RESET)
489  {
490  stable ++;
491  }
492  retry --;
493  }
494  while(stable <5 && retry>0);
495  if(!retry)
496  goto Error;
497 
498  return M24SR_STATUS_SUCCESS;
499 
500  default :
501  return M24SR_ERROR_DEFAULT;
502  }
503 
504 Error :
505  return M24SR_ERROR_DEFAULT;
506 }
